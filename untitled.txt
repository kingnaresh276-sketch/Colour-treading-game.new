<script>
let balance = 1000;
let betChoice = {type:"color", value:"red"};

function select(type, value){
  betChoice = {type, value};
  alert("Selected: " + type + " = " + value);
}

function spin(){
  let number = Math.floor(Math.random()*10);
  let color = number <= 3 ? "Red" : number <= 7 ? "Green" : "Violet";
  let size = number <= 4 ? "Small" : "Big";
  return {number, color, size};
}

function play(){
  let bet = Number(document.getElementById("bet").value);

  if(bet > balance){
    alert("Not enough balance");
    return;
  }

  let result = spin();
  document.getElementById("num").innerText = result.number;
  document.getElementById("color").innerText = result.color;
  document.getElementById("size").innerText = result.size;

  let win = false;

  if(betChoice.type === "color" && betChoice.value.toLowerCase() === result.color.toLowerCase()) win = true;
  if(betChoice.type === "size" && betChoice.value.toLowerCase() === result.size.toLowerCase()) win = true;
  if(betChoice.type === "number" && betChoice.value == result.number) win = true;

  if(win){
    balance += bet;
    alert("ðŸŽ‰ You Win! +â‚¹" + bet);
  } else {
    balance -= bet;
    alert("ðŸ˜¢ You Lose! -â‚¹" + bet);
  }

  document.getElementById("balance").innerText = balance;

  let li = document.createElement("li");
  li.innerText = `Bet: â‚¹${bet} | Pick: ${betChoice.type}:${betChoice.value} | Result: ${result.number}-${result.color}-${result.size}`;
  document.getElementById("history").prepend(li);
}
